\begin{frame}{Sub-project 1: Offspring loss}

\textbf{How does offspring loss affect parents?} 
\vspace{0.5cm}

 \begin{enumerate}
     \item Causal analysis to expand my descriptive work on offspring loss
     \item Increasing data availability in North and South (e.g., PSID, CHARLS, DHS, Twitter, online surveys)
    %  \item Gender dimension
 \end{enumerate}
\end{frame}


\begin{frame}{Sub-project 1: Offspring loss}

\textbf{Expected contributions:} 
\vspace{0.5cm}

     \begin{enumerate}
         \item Differential effects by contextual factors (e.g., single child)
         \item Links to outsiders to identify priorities
        \item  Support to bereaved parents (at which age does loss happen?)

     \end{enumerate}

\end{frame}


\begin{frame}{Sub-project 2: Kinship transitions}

\textbf{How does demographic change unfold from the perspective of kin?} 
\vspace{0.5cm}

 \begin{enumerate}
    % \item \textit{How does demographic change unfold from the perspective of kin?}
    \item Build on my work on sandwich generation
    \item Register data, crowd-sourced genealogies, linked censuses
    \item Novel formal methods and simulations techniques
 \end{enumerate}
 \end{frame}


\begin{frame}{Sub-project 2: Kinship transitions}

\textbf{Expected contributions:} 
\vspace{0.5cm}

    \begin{enumerate}
        %  \item Relationship between demographic change, kin availability, and generational overlap
         \item Disentangle structure from function
         \item Map kinship to households
         \item `Slow' population aging in North vs `rapid' demographic change in South
         \item Policy: burden of care
        %  \item Within-country heterogeneity and regional inequalities
     \end{enumerate}
     
\end{frame}


\begin{frame}{Sub-project 3: Consequences of mortality crises}

\textbf{How do kinship networks amplify the effects of mortality crises in the short- and long term?} 
\vspace{0.5cm}

 \begin{enumerate}
     \item Build on my previous work on the demography of genocide 
    %  \item No lack of mortality crises
     \item Strong (but not exclusive) focus on Covid-19\footnote{Martinez-Folgar, K., \textcolor{blue}{Alburez-Gutierrez, D.}, et al. (2021). ``Excess Mortality during the COVID-19 Pandemic in Guatemala.'' \textbf{American Journal of Public Health}, 0:e1-e8.}
     \item Demographic data repositories (e.g., HMD, STMF, HFD, WPP, COVerAGE-DB)

 \end{enumerate}
 
\end{frame}


\begin{frame}{Sub-project 3: Consequences of mortality crises}

\textbf{Expected contributions:} 
\vspace{0.5cm}

	 \begin{enumerate}
         \item Theories linking social change to population dynamics and composition
         \item Scale up to national populations
        \item How does excess mortality translate into kin loss?
         \item Policies to support bereaved relatives (Covid-19\footnote{Nepomuceno, M., Acosta, E., \textcolor{blue}{Alburez-Gutierrez, D.}, et al. (2020). ``Besides population age-structure, health and other demographic factors can contribute to understanding the COVID-19 burden." \textbf{Proceedings of the National Academy of Sciences}, 117(25).})
     \end{enumerate}
     
\end{frame}
