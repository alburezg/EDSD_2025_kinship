\begin{frame}{Definitions (1)\fcite{alburez-gutierrez_kinship_2022}}
   \begin{exampleblock}{Kinship}
        Social relationships that bind individuals together through culturally shared definitions of relatedness on biological, legal, or normative grounds, ultimately constituting family systems.
   \end{exampleblock}
   
      \vspace{0.5cm}
   
    \begin{alertblock}{Family}
        More narrow group of kin given special privilege which, among other things, organize the provision of support, socialization, and social placement of its members.
  \end{alertblock}

\end{frame}

\begin{frame}{The role of kinship in human societies}
   \begin{enumerate}
       \item Socialisation, protection, and sustenance
       \item Inter-generational solidarity: exchanges and bequests
       \item Social structure and identity
       \item Early-life conditions $\rightarrow$ later-life outcomes
   \end{enumerate}
\end{frame}



\begin{frame}{Definitions (2)\fcite{alburez-gutierrez_kinship_2022}}
   \begin{exampleblock}{Kinship demography}
        The study of family networks, their structures and dynamics from a demographic perspective and using demographic methods.
   \end{exampleblock}
\end{frame}

\begin{frame}{Kinship as a demographic human universal}
    \begin{enumerate}
        \item All humans are born 
        \item All humans die
        \item All humans are embedded in kinship structures\fcite{caswell_formal_2019}
        \item No particular family configuration is universal or stable
    \end{enumerate}
\end{frame}

\begin{frame}{The \textit{Strong Ergodic Principle}\fcite{sharpe_problem_1911}}

    \say{A closed population with unchanging mortality and fertility rates has an \textbf{implicit age structure}.}
    % intrinsic rate of natural increase $r$ and 
    \vspace{0.3cm}
    
    The proportion of the population aged $x$ to $x+n$ in a stable population is\fcite{wachter_essential_2014}:
    
    \begin{equation}
      \frac{_{n}K_{x}}{_{\infty}K_{0}} = b\frac{_{n}L_{x}}{l_{0}}e^{-rx}
    \end{equation}

\end{frame}

\begin{frame}{Demographic foundations of lived experience of kin loss}

    \textbf{Child Death} $CD_{(a,c)}$ is the cumulative number of child deaths experienced by a woman in cohort $c$, conditional on her survival to age \textit{a}:
    
    \begin{equation*}
    Children\:born_{(a,c)} = Children\:surviving_{(a,c)} + Children\:died_{(a,c)}
    \end{equation*}
  
\end{frame}


\begin{frame}{Demographic foundations of lived experience of kin loss}

\textbf{Child Death} $CD_{(a,c)}$ is the cumulative number of child deaths experienced by a woman in cohort $c$, conditional on her survival to age \textit{a}:

\begin{equation*}
Children\:died_{(a,c)} = Children\:born_{(a,c)} - Children\:surviving_{(a,c)}
\end{equation*}
  
\end{frame}


% \begin{frame}{The Child Death identity}
\begin{frame}{Demographic foundations of lived experience of kin loss}

\textbf{Child Death} $CD_{(a,c)}$ is the cumulative number of child deaths experienced by a woman in cohort $c$, conditional on her survival to age \textit{a}\fcite{goodman_family_1974}:

\begin{equation*}
\label{eq:CD}
\underbrace{CD_{(a,c)}}_{\text{Children died}} = \underbrace{\sum_{x=15}^{x=a} {_1F_{(x,c)}}}_{\text{Children born}}-\underbrace{\sum_{x=15}^{x=a} {_1F_{(x,c)}} l_{(a-x,c+x)}}_{\text{Children surviving}}
\end{equation*}
  
 where

\begin{itemize}
  \item $_1F_{(x,c)}$ are age-specific fertility rates for mothers
  \item $l_{(a-x,c+x)}$ are age-specific mortality probabilities for mother's children
\end{itemize}

\end{frame}

\begin{frame}{Mortality from a kinship perspective}
    \begin{enumerate}
        % \item Mortality and bereavement are different
        \item Mortality $\Rightarrow$ kin loss $\Rightarrow$ breavement $\Rightarrow$ memory
        \item Mortality crises `accelerate' the experience of bereavement
        \item Implications for individuals and populations
    \end{enumerate}
    
    \begin{equation*}
        \text{Kin Loss} \approx \text{Mortality} \times \text{Kinship Structure}
    \end{equation*}

\end{frame}
